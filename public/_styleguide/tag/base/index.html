classなし

data-src="../../../../../lib/img/dummy/yamato.png"

<img
data-src="../../../../../lib/img/dummy/yamato.png"
alt="dummy"
class="u-img__max">

data-src="../../../../../lib/img/dummy/yamato.png"
alt="dummy"
